Laboratório

Variáveis e operadores

// declaração de variáveis
var x;            //declaração da variável
var x1;           //nome: letras, números, _
var var;          //certas palavras são reservadas
var function;     //inválido! - function é palavra reservada
x = 4;            // = atribui o valor à variável
var y = 4;        // você pode declarar e atribuir valor em um único comando
var abC; var aBc; //nome é sensível a maísculas/ minúsculas
var enemyLife;    //começa com letra minúscula, letras maiúsculas separam palavras (camel case)

// operações matemáticas
var x = 1 + 3; //soma
var y = 1 - 3; //subtração
var z = 4 * 5; //multiplicação
var i = 20 / 5; //divisão
