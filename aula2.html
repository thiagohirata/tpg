Variáveis e operadores

// declaração de variáveis
var x;            //declaração da variável
var x1;           //identificador, pode conter números, letras, '_' e '$' (tem mais regras)
var var;          //certas palavras são reservadas - não chame uma variável de "var"
var function;     //inválido! - function é palavra reservada
var var1;         //var1 é válido
x = 4;            // = atribui o valor à variável
var y = 4;        // você pode declarar e atribuir valor em um único comando
var abC; var aBc; //identificador é sensível a maísculas/ minúsculas
var enemyLife;    //camel case: começa com letra minúscula, letras maiúsculas separam palavras

// operações matemáticas
var x = 1 + 3;   //soma
var y = 1 - 3;   // subtração
var z = 4 * 5;   // multiplicação
var i = 20 / 5;  // divisão
var j = 14 % 10; // resto da divisão (4)
x += 4; // mesma coisa que x = x + 4;
x -= 4; // mesma coisa que x = x - 4;
x++; y--; // igual a x = x + 1; y = y - 1;

Exemplos

size(555,300);
var lado = 40;
var posX = 50;
rect(posX, 150, lado, lado); // igual a rect(50, 150, 40, 40);
posX = posX + 50;            // posX agora é igual a 100
rect(posX, 150, lado, lado); // igual a rect(150, 150, 40, 40);
posX = posX + 50;     
rect(posX, 150, lado, lado);
posX += 50; //outra forma de escrever a mesma coisa     
rect(posX, 150, lado, lado);
posX += 50;      
rect(posX, 150, lado, lado);

size(555,300);
var lado = 40;
var posX = 50;
rect(posX,       50, lado, lado); // igual a rect(50, 150, 40, 40);
//nas linhas abaixo, posX não muda de valor - continua igual a 50
rect(posX + 50,  50, lado, lado); // rect(100, 50, 40, 40) 
rect(posX + 100, 50, lado, lado); // rect(150, 50, 40, 40) 
rect(posX + 150, 50, lado, lado);
rect(posX + 200, 50, lado, lado);
rect(posX + 250, 50, lado, lado);
