Laboratório

if ... else, operadores de comparação e operadores lógicos

// operadores de comparação
// == - valor igual
5 == 5;         // Verdadeiro   
5 == 8;         // Falso 
5 == '5';       // Verdadeiro (no JavaScript)
// === - igual e mesmo tipo
5 === 5;        // Verdadeiro   
5 === 8;        // Falso 
5 === '5';      // Falso
// != - valor diferente
5 != 5;         // Falso   
5 != 8;         // Verdadeiro 
5 != '5';       // Falso (no JavaScript)
// != - valor ou tipo diferente
5 !== 5;        // Falso   
5 !== 8;        // Verdadeiro 
5 !== '5';      // Verdadeiro
// > - maior que
5 > 5;        // Falso   
5 > 4;        // Verdadeiro
// >= - maior que ou igual a
5 >= 5;        // Verdadeiro   
5 >= 4;        // Verdadeiro
// < - menor que
5 < 5;        // Falso   
5 < 4;        // Falso
// <= - menor que ou igual a
5 <= 5;        // Verdadeiro   
5 <= 4;        // Falso

// if
if ( /* aqui vai a condição */ ) {
	//trecho de código que é executado se a condição for verdadeira
}
//if ... else
if (/*condição */) {
	//executa isto se condição é verdadeira
} else {
	//executa isto se condição é falsa
}

var entradaPermitida = idade >= 18;
if (entradaPermitida) {
	abrirPagina('http://paginaadulta.com.br');
} else {
	abrirPagina('http://disney.com');
	tocarSom('choro.wav');
}

//if ... else if ... if
if ( ano > 2010 ) {
	enviarWhatsApp(mensagem);
} else if ( ano > 2004 ) {
	enviarFacebook(mensagem);
} else if ( ano > 1995 ) {
	enviarEmail(mensagem);
} else {
	enviarCarta(mensagem);
}
// operadores lógicos
// && - e lógico
if (idade >= 4 && idade <= 13) {
	faixaEtaria = 'Criança';
}
// || - ou lógico
if (faixaEtaria == 'Criança' || faixaEtaria == 'Bebê' || faixaEtaria == 'Adolescente') {
	podeAssistirAoFilmeDoDeadPool = false;
}
// ! - negação
if (!podeAssistirAoFilmeDoDeadPool) {
	tocarSom('choro.wav');
}

Complete o código abaixo para incluir as regras do jogo da velha

size(400, 400);
//tabuleiro de jogo da velha
